The parser for embedding YT URLs ignores timecodes in them (deadly when people are intentionally linking critical parts inside long clips).

This replacement code for the Media object's toggleYouTube function (inside extension.compiled.js) makes sure the parser grabs and passes them without altering expected embed/remove behavior:

toggleYouTube:function(a){var b,c,t;t='';timeYT=/((#)|(\&))t=(\d+m)?(\d+s)?/;"Remove"==a.textContent?(a.parentNode.removeChild(a.nextElementSibling),a.textContent="Embed"):(b=a.previousElementSibling.textContent.match(this.toggleYT))&&(b=b[1])?(t=timeYT.test(a.previousElementSibling.textContent)?a.previousElementSibling.textContent.match(timeYT)[0]:'',c=document.createElement("div"),c.className="media-embed",c.innerHTML='<iframe src="//www.youtube.com/embed/'+encodeURIComponent(b)+t+'" width="640" height="360" frameborder="0"></iframe>',a.parentNode.insertBefore(c,a.nextElementSibling),a.textContent="Remove"):a.textContent="Error"}

Ideally timeYT would be scoped to this.timeYT in the Media object's init function, this is just a proof of concept.
